
(defun checkNP (parse speechPart)
	(prog ()
		(setf article (getToken parse))
		(if (isa article 'article)
			(setf noun (getToken parse))
			(setf noun article)
		)
		(if (not (isa noun 'noun))
			(return NIL)
		)
		(if (isa article 'article)
			(putp speechPart parse (list article noun))
		)
		(if (isa article 'noun)
			(putp speechPart parse (list noun))
		)
		(return T)
	)
)
